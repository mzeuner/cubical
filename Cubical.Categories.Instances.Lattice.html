{-# OPTIONS --safe #-}
module Cubical.Categories.Instances.Lattice where

open import Cubical.Foundations.Prelude

open import Cubical.Algebra.Lattice

open Category

LatticeCategory : ∀ {ℓ} (L : Lattice ℓ) → Category ℓ ℓ
